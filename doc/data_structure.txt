	class cdraft << (M,orchid) >> {
		CIPHERID		cipherid
		--
	   -DRAFTID				draftid
	    uint16				version
	    uint16				no
		bool				formal
		string				name
		vector<string>		tags
		vector<name>		editors
		ipfskey				hash
		uint16				nofapproval
		vector<name>		approvers
		vector<name>		approved
		vector<TASKID>		tasklist
		vector<TOKENID>	tokenlist
		--
		key2		draftid + version + no
	}

	class cformal << (M,orchid) >> {
		name			"myphersystem"
		--
	   -CIPHERID		ciperid

}
